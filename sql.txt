CREATE TABLE category (
	cid   int         NOT NULL AUTO_INCREMENT,
	cname VARCHAR(50) NOT NULL,
	PRIMARY KEY (cid)
);

CREATE TABLE transaction (
    tid          int NOT NULL AUTO_INCREMENT,
    tamount      int NOT NULL,
    tdescription VARCHAR(100),
    tdate        DATETIME,
    PRIMARY KEY (tid)
);

CREATE TABLE user (
	uid     int NOT NULL AUTO_INCREMENT,
	balance int NOT NULL,
	PRIMARY KEY (uid)
);

CREATE TABLE budget (
	bid     int NOT NULL AUTO_INCREMENT,
	bamount int NOT NULL,
	PRIMARY KEY (bid)
);

CREATE TABLE makes (
	uid int NOT NULL,
	tid int NOT NULL,
	FOREIGN KEY (uid) REFERENCES user        (uid) ON DELETE CASCADE,
	FOREIGN KEY (tid) REFERENCES transaction (tid) ON DELETE CASCADE,
	PRIMARY KEY (uid, tid)
);

CREATE TABLE belongs_to (
	cid int NOT NULL,
	tid int NOT NULL,
	FOREIGN KEY (cid) REFERENCES category     (cid) ON DELETE CASCADE,
	FOREIGN KEY (tid) REFERENCES transactions (tid) ON DELETE CASCADE,
	PRIMARY KEY (cid, tid)
);

CREATE TABLE sets (
	uid int NOT NULL,
	bid int NOT NULL,
	FOREIGN KEY (uid) REFERENCES user   (uid) ON DELETE CASCADE,
	FOREIGN KEY (bid) REFERENCES budget (bid) ON DELETE CASCADE,
	PRIMARY KEY (uid, bid)
);

CREATE TABLE of (
	cid int NOT NULL,
	bid int NOT NULL,
	FOREIGN KEY (cid) REFERENCES category (cid) ON DELETE CASCADE,
	FOREIGN KEY (bid) REFERENCES budget   (bid) ON DELETE CASCADE,
	PRIMARY KEY (cid, bid)
);
